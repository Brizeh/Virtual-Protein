import json import pandas as pd from bokeh.plotting import figure, show, output_file from bokeh.models import ( ColumnDataSource, Select, CustomJS, HoverTool, LabelSet, Div, TapTool, TextInput, Button, DataRange1d, Span, WheelZoomTool ) from bokeh.layouts import column, row protWeight = 1000 def price_per_prot(food, protWeight): price_per_100 = food.get("price") / food.get("weight") * 100 k = price_per_100 / food.get("protein") return k * protWeight def calories_per_prot(food, protWeight): return food.get("calories") / food.get("protein") * protWeight def carbs_per_prot(food, protWeight): return food.get("carbs") / food.get("protein") * protWeight def fat_per_prot(food, protWeight): return food.get("fat") / food.get("protein") * protWeight def weight_per_prot(food, protWeight): return 100 * protWeight / food.get("protein") with open("foods.json", "r", encoding="utf-8") as f: all_foods = json.load(f) names = [food for food in all_foods.keys()] weight_vals = [weight_per_prot(food, protWeight) for food in all_foods.values()] price_vals = [price_per_prot(food, protWeight) for food in all_foods.values()] cal_vals = [calories_per_prot(food, protWeight) for food in all_foods.values()] fat_vals = [fat_per_prot(food, protWeight) for food in all_foods.values()] carb_vals = [carbs_per_prot(food, protWeight) for food in all_foods.values()] images = [food.get("image") for food in all_foods.values()] links = [food.get("link") for food in all_foods.values()] colors = ["navy"] * len(names) source = ColumnDataSource(dict( Food=names, Weight=weight_vals, Price=price_vals, Calories=cal_vals, Fat=fat_vals, Carbs=carb_vals, Image=images, Link=links, Color=colors )) p = figure( width=900, height=600, title="Food Comparison per 1kg of Protein : Click on points for more info", x_axis_label="Price", y_axis_label="Calories", tools="pan,wheel_zoom,box_zoom,reset", x_range=DataRange1d(start=0), y_range=DataRange1d(start=0) ) p.toolbar.active_scroll = p.select_one(WheelZoomTool) p.x_range.js_on_change("start", CustomJS(code=""" if (this.start < 0) { const shift = -this.start this.start = 0 this.end += shift } """)) p.y_range.js_on_change("start", CustomJS(code=""" if (this.start < 0) { const shift = -this.start this.start = 0 this.end += shift } """)) x_axis_zero = Span( location=0, dimension="width", line_color="black", line_width=3 ) y_axis_zero = Span( location=0, dimension="height", line_color="black", line_width=3 ) p.add_layout(x_axis_zero) p.add_layout(y_axis_zero) scatter = p.circle( "Price", "Calories", size=10, color="Color", source=source ) labels = LabelSet( x="Price", y="Calories", text="Food", source=source, x_offset=5, y_offset=5 ) p.add_layout(labels) hover = HoverTool(tooltips="""
@Food
Values for 1kg of protein
Price : @Price{0.00}€
Weight : @Weight{0}g
Calories : @Calories{0}kcal
Fat : @Fat{0}g
Carbs : @Carbs{0}g
""") p.add_tools(hover) # Axis selectors options = ["Price", "Calories", "Fat", "Carbs"] x_select = Select(title="X Axis", value="Price", options=options) y_select = Select(title="Y Axis", value="Calories", options=options) axis_callback = CustomJS( args=dict( scatter=scatter, labels=labels, x_select=x_select, y_select=y_select, xaxis=p.xaxis[0], yaxis=p.yaxis[0], ), code=""" const x = x_select.value const y = y_select.value scatter.glyph.x = {field: x} scatter.glyph.y = {field: y} labels.x = {field: x} labels.y = {field: y} const axis_labels = { "Price": "Price (€ for 1kg protein)", "Calories": "Calories (kcal for 1kg protein)", "Fat": "Fat (g for 1kg protein)", "Carbs": "Carbs (g for 1kg protein)" } xaxis.axis_label = axis_labels[x] yaxis.axis_label = axis_labels[y] """ ) x_select.js_on_change("value", axis_callback) y_select.js_on_change("value", axis_callback) # Click info card info_div = Div(width=260, height=260) tap_callback = CustomJS(args=dict(source=source, div=info_div), code=""" const i = source.selected.indices[0] if (i == null) return const d = source.data const food = d['Food'][i] const weight = d['Weight'][i].toFixed(0) const price = d['Price'][i].toFixed(2) const cal = d['Calories'][i].toFixed(0) const fat = d['Fat'][i].toFixed(0) const carbs = d['Carbs'][i].toFixed(0) const img = d['Image'][i] const link = d['Link'][i] let img_html = "" let link_html = "" if (img) { img_html = `

` } if (link) { link_html = `View product` } div.text = `
${img_html} ${food}

Values for 1kg of protein
Price : ${price}€
Weight : ${weight}g
Calories : ${cal}kcal
Fat : ${fat}g
Carbs : ${carbs}g

${link_html}
` """) p.add_tools(TapTool(callback=tap_callback)) # Custom food inputs food_input = TextInput(title="Add new food here : Food name", value="Custom food") price_input = TextInput(title="Total price (€)") weight_input = TextInput(title="Total weight (g)") protein_input = TextInput(title="Protein for 100g") calories_input = TextInput(title="Calories for 100g") fat_input = TextInput(title="Fat for 100g") carbs_input = TextInput(title="Carbs for 100g") add_button = Button(label="Add Food", button_type="success") add_callback = CustomJS( args=dict( source=source, food_input=food_input, price_input=price_input, weight_input=weight_input, protein_input=protein_input, calories_input=calories_input, fat_input=fat_input, carbs_input=carbs_input ), code=""" const food = food_input.value const price = parseFloat(price_input.value) const weight = parseFloat(weight_input.value) const prot = parseFloat(protein_input.value) const cal = parseFloat(calories_input.value) const fat = parseFloat(fat_input.value) const carbs = parseFloat(carbs_input.value) if (isNaN(price) || isNaN(weight) || isNaN(prot)) { alert("Please fill price, weight and protein") return } const protWeight = 1000 const price100 = price / weight * 100 const k = price100 / prot const weightProt = 100 * protWeight / prot const priceProt = k * protWeight const calProt = cal / prot * protWeight const fatProt = fat / prot * protWeight const carbProt = carbs / prot * protWeight source.stream({ Food: [food], Weight: [weightProt], Price: [priceProt], Calories: [calProt], Fat: [fatProt], Carbs: [carbProt], Image: ["https://wiki.guildwars2.com/images/thumb/a/ab/Bladesworn_icon_%28highres%29.png/160px-Bladesworn_icon_%28highres%29.png"], Link: ["https://wiki.guildwars2.com/wiki/Bladesworn"], Color: ["red"] }) """ ) add_button.js_on_click(add_callback) inputs = column( food_input, price_input, weight_input, protein_input, calories_input, fat_input, carbs_input, add_button, width=250 ) layout = column( row(x_select, y_select), row(p, inputs, info_div) ) output_file("protein_comparison_bokeh.html") show(layout)
